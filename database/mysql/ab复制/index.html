    abеӨҚеҲ¶

    MySQL зҡ„ AB еӨҚеҲ¶пјҲдё»д»ҺеӨҚеҲ¶пјүжҳҜдёҖз§Қеёёи§Ғзҡ„ж•°жҚ®еҗҢжӯҘжңәеҲ¶пјҢз”ЁдәҺеңЁдё»жңҚеҠЎеҷЁпјҲAпјүе’Ңд»ҺжңҚеҠЎеҷЁпјҲBпјүд№Ӣй—ҙеӨҚеҲ¶ж•°жҚ®пјҢд»Ҙе®һзҺ°й«ҳеҸҜз”ЁжҖ§е’ҢиҙҹиҪҪеқҮиЎЎгҖӮд»ҘдёӢжҳҜйғЁзҪІ MySQL AB еӨҚеҲ¶зҡ„иҜҰз»ҶжӯҘйӘӨпјҡ

    1. зҺҜеўғеҮҶеӨҮ

    • дёӨеҸ°жңҚеҠЎеҷЁпјҡдё»жңҚеҠЎеҷЁпјҲAпјүе’Ңд»ҺжңҚеҠЎеҷЁпјҲBпјүгҖӮ
    • е·Іе®үиЈ…е№¶иҝҗиЎҢ MySQL ж•°жҚ®еә“гҖӮ
    • зЎ®дҝқдёӨеҸ°жңҚеҠЎеҷЁеҸҜд»ҘйҖҡиҝҮзҪ‘з»ңзӣёдә’йҖҡдҝЎгҖӮ

    2. дё»жңҚеҠЎеҷЁй…ҚзҪ®пјҲAпјү

    дҝ®ж”№ MySQL й…ҚзҪ®ж–Үд»¶

    зј–иҫ‘дё»жңҚеҠЎеҷЁзҡ„ MySQL й…ҚзҪ®ж–Үд»¶пјҲдёҖиҲ¬дҪҚдәҺ /etc/my.cnf жҲ– /etc/mysql/my.cnfпјүпјҢж·»еҠ д»ҘдёӢеҶ…е®№пјҡ

    [mysqld]
server-id=1                 # е”ҜдёҖIDпјҢдё»д»ҺжңҚеҠЎеҷЁеҝ…йЎ»дёҚеҗҢ
log-bin=mysql-bin           # ејҖеҗҜдәҢиҝӣеҲ¶ж—Ҙеҝ—
binlog-do-db=your_database  # йңҖиҰҒеӨҚеҲ¶зҡ„ж•°жҚ®еә“
binlog-ignore-db=mysql      # дёҚйңҖиҰҒеӨҚеҲ¶зҡ„ж•°жҚ®еә“пјҲеҸҜйҖүпјү

    йҮҚеҗҜ MySQL жңҚеҠЎ

    sudo systemctl restart mysqld

    еҲӣе»әеӨҚеҲ¶з”ЁжҲ·

    зҷ»еҪ• MySQLпјҢеҲӣе»әдёҖдёӘз”ЁжҲ·з”ЁдәҺеӨҚеҲ¶пјҡ

    CREATE USER 'replica_user'@'%' IDENTIFIED BY 'password';
GRANT REPLICATION SLAVE ON *.* TO 'replica_user'@'%';
FLUSH PRIVILEGES;

    иҺ·еҸ–дё»жңҚеҠЎеҷЁзҡ„дәҢиҝӣеҲ¶ж—Ҙеҝ—дҝЎжҒҜ

    SHOW MASTER STATUS;
    и®°дёӢд»ҘдёӢдҝЎжҒҜпјҡ - FileпјҲдәҢиҝӣеҲ¶ж—Ҙеҝ—ж–Үд»¶еҗҚпјү - PositionпјҲдәҢиҝӣеҲ¶ж—Ҙеҝ—дҪҚзҪ®пјү

    3. д»ҺжңҚеҠЎеҷЁй…ҚзҪ®пјҲBпјү

    дҝ®ж”№ MySQL й…ҚзҪ®ж–Үд»¶

    зј–иҫ‘д»ҺжңҚеҠЎеҷЁзҡ„ MySQL й…ҚзҪ®ж–Үд»¶пјҢж·»еҠ д»ҘдёӢеҶ…е®№пјҡ

    [mysqld]
server-id=2                 # е”ҜдёҖIDпјҢдёҺдё»жңҚеҠЎеҷЁдёҚеҗҢ
relay-log=relay-log         # ејҖеҗҜдёӯз»§ж—Ҙеҝ—пјҲеҸҜйҖүпјү
read-only=1                 # и®ҫзҪ®еҸӘиҜ»пјҲеҸҜйҖүпјҢзЎ®дҝқд»Һеә“дёҚдјҡиў«ж„ҸеӨ–еҶҷе…Ҙпјү

    йҮҚеҗҜ MySQL жңҚеҠЎ

    sudo systemctl restart mysqld

    й…ҚзҪ®еӨҚеҲ¶

    зҷ»еҪ•д»ҺжңҚеҠЎеҷЁзҡ„ MySQLпјҢжү§иЎҢд»ҘдёӢе‘Ҫд»Өпјҡ

    CHANGE MASTER TO
  MASTER_HOST='дё»жңҚеҠЎеҷЁIPең°еқҖ',
  MASTER_USER='replica_user',
  MASTER_PASSWORD='password',
  MASTER_LOG_FILE='дё»жңҚеҠЎеҷЁзҡ„File',
  MASTER_LOG_POS=дё»жңҚеҠЎеҷЁзҡ„Position;

    еҗҜеҠЁеӨҚеҲ¶

    START SLAVE;

    жЈҖжҹҘеӨҚеҲ¶зҠ¶жҖҒ

    SHOW SLAVE STATUS\G;
    зЎ®дҝқд»ҘдёӢдёӨйЎ№жҳҫзӨәдёә Yesпјҡ - Slave_IO_Running: Yes - Slave_SQL_Running: Yes

    4. жөӢиҜ•дё»д»ҺеӨҚеҲ¶

    • еңЁдё»жңҚеҠЎеҷЁдёҠеҗ‘ your_database жҸ’е…Ҙж•°жҚ®гҖӮ
    • еңЁд»ҺжңҚеҠЎеҷЁдёҠжҹҘиҜў your_databaseпјҢзЎ®дҝқж•°жҚ®еҗҢжӯҘжҲҗеҠҹгҖӮ

    жіЁж„ҸдәӢйЎ№

    1. ж•°жҚ®дёҖиҮҙжҖ§пјҡеңЁи®ҫзҪ®д»Һеә“еүҚпјҢзЎ®дҝқд»Һеә“е’Ңдё»еә“ж•°жҚ®дёҖиҮҙпјҢеҸҜйҖҡиҝҮеӨҮд»ҪжҒўеӨҚгҖӮ
    2. йҳІзҒ«еўҷи®ҫзҪ®пјҡзЎ®дҝқ 3306 з«ҜеҸЈејҖж”ҫпјҢе…Ғи®ёд»Һеә“и®ҝй—®дё»еә“гҖӮ
    3. й«ҳеҸҜз”ЁжҖ§пјҡдёәйҒҝе…ҚеҚ•зӮ№ж•…йҡңпјҢеҸҜд»Ҙз»“еҗҲ MHAгҖҒKeepalived жҲ–е…¶д»–е·Ҙе…·е®һзҺ°ж•…йҡңиҮӘеҠЁеҲҮжҚўгҖӮ

    й…ҚзҪ®еӨҡдёӘд»ҺжңҚеҠЎеҷЁ

    еңЁ MySQL AB еӨҚеҲ¶дёӯпјҢж·»еҠ еӨҡдёӘд»ҺжңҚеҠЎеҷЁйқһеёёз®ҖеҚ•гҖӮд»Һеә“д№Ӣй—ҙжІЎжңүзӣҙжҺҘе…ізі»пјҢйғҪжҳҜзӢ¬з«ӢиҝһжҺҘеҲ°дё»жңҚеҠЎеҷЁзҡ„пјҢдё»жңҚеҠЎеҷЁдјҡеҗ‘жүҖжңүд»ҺжңҚеҠЎеҷЁеҗҢжӯҘдәҢиҝӣеҲ¶ж—Ҙеҝ—гҖӮд»ҘдёӢжҳҜй…ҚзҪ®дёӨеҸ°д»ҺжңҚеҠЎеҷЁзҡ„жӯҘйӘӨпјҡ

    1. зҺҜеўғиҜҙжҳҺ

    • дё»жңҚеҠЎеҷЁ (A)пјҡиҙҹиҙЈеӨ„зҗҶеҶҷж“ҚдҪңе’ҢжҸҗдҫӣдәҢиҝӣеҲ¶ж—Ҙеҝ—гҖӮ
    • д»ҺжңҚеҠЎеҷЁ (B е’Ң C)пјҡиҙҹиҙЈд»Һдё»жңҚеҠЎеҷЁеҗҢжӯҘж•°жҚ®гҖӮ

    еҒҮи®ҫпјҡ - дё»жңҚеҠЎеҷЁ IPпјҡ192.168.1.100 - д»ҺжңҚеҠЎеҷЁ B IPпјҡ192.168.1.101 - д»ҺжңҚеҠЎеҷЁ C IPпјҡ192.168.1.102 - еӨҚеҲ¶з”ЁжҲ·пјҡreplica_userпјҢеҜҶз Ғпјҡpassword

    2. дё»жңҚеҠЎеҷЁй…ҚзҪ®

    дё»жңҚеҠЎеҷЁзҡ„й…ҚзҪ®дёҺеҚ•д»Һеә“ж—¶зӣёеҗҢпјҢж— йңҖйўқеӨ–и°ғж•ҙпјҡ 1. зЎ®дҝқдё»жңҚеҠЎеҷЁзҡ„ server-id е”ҜдёҖгҖӮ 2. ејҖеҗҜдәҢиҝӣеҲ¶ж—Ҙеҝ—пјҡ 

    [mysqld]
server-id=1
log-bin=mysql-bin
    3. еҲӣе»әдёҖдёӘеӨҚеҲ¶з”ЁжҲ·пјҲеҰӮд№ӢеүҚжүҖиҝ°пјүгҖӮ

    еҰӮжһңд№ӢеүҚе·Іе®ҢжҲҗдё»д»ҺеӨҚеҲ¶пјҢеҸӘйңҖиҰҒзЎ®дҝқдё»еә“е·ІжӯЈеёёиҝҗиЎҢгҖӮ

    3. д»ҺжңҚеҠЎеҷЁ B е’Ң C й…ҚзҪ®

    3.1 дҝ®ж”№й…ҚзҪ®ж–Үд»¶

    еңЁжҜҸдёӘд»ҺжңҚеҠЎеҷЁпјҲB е’Ң Cпјүзҡ„ MySQL й…ҚзҪ®ж–Үд»¶пјҲ/etc/my.cnf жҲ– /etc/mysql/my.cnfпјүдёӯпјҢи®ҫзҪ®еҗ„иҮӘзҡ„ server-id е’Ңеҝ…иҰҒзҡ„еҸӮж•°пјҡ

    д»ҺжңҚеҠЎеҷЁ B й…ҚзҪ®пјҡ 

    [mysqld]
server-id=2                 # д»Һеә“ B зҡ„е”ҜдёҖ ID
relay-log=relay-log         # дёӯз»§ж—Ҙеҝ—ж–Үд»¶
read-only=1                 # и®ҫзҪ®еҸӘиҜ»пјҲеҸҜйҖүпјү

    д»ҺжңҚеҠЎеҷЁ C й…ҚзҪ®пјҡ 

    [mysqld]
server-id=3                 # д»Һеә“ C зҡ„е”ҜдёҖ ID
relay-log=relay-log         # дёӯз»§ж—Ҙеҝ—ж–Үд»¶
read-only=1                 # и®ҫзҪ®еҸӘиҜ»пјҲеҸҜйҖүпјү

    3.2 йҮҚеҗҜ MySQL жңҚеҠЎ

    йҮҚеҗҜжҜҸдёӘд»ҺжңҚеҠЎеҷЁзҡ„ MySQL жңҚеҠЎд»ҘеҠ иҪҪж–°й…ҚзҪ®пјҡ

    sudo systemctl restart mysqld

    3.3 и®ҫзҪ®д»Һеә“еҗҢжӯҘ

    еңЁжҜҸдёӘд»ҺжңҚеҠЎеҷЁпјҲB е’Ң Cпјүдёӯзҷ»еҪ• MySQLпјҢе№¶й…ҚзҪ®еҗҢжӯҘпјҡ

    1. йҖҡиҝҮдё»жңҚеҠЎеҷЁзҡ„еӨҮд»ҪеҲқе§ӢеҢ–д»Һеә“ еҰӮжһңд»Һеә“жңӘеҲқе§ӢеҢ–ж•°жҚ®пјҢйңҖйҖҡиҝҮдё»еә“еӨҮд»ҪжҒўеӨҚж•°жҚ®дёҖиҮҙжҖ§пјҡ

    2. еңЁдё»еә“жү§иЎҢеӨҮд»Ҫпјҡ 

      mysqldump -u root -p --all-databases --master-data > backup.sql

    3. е°Ҷ backup.sql жӢ·иҙқеҲ°д»Һеә“пјҢеҜје…Ҙж•°жҚ®пјҡ 

      mysql -u root -p < backup.sql

    4. и®ҫзҪ®еӨҚеҲ¶еҸӮж•° зҷ»еҪ•д»Һеә“ B е’Ңд»Һеә“ CпјҢеҲҶеҲ«жү§иЎҢд»ҘдёӢе‘Ҫд»Өпјҡ 

      CHANGE MASTER TO
  MASTER_HOST='192.168.1.100',
  MASTER_USER='replica_user',
  MASTER_PASSWORD='password',
  MASTER_LOG_FILE='mysql-bin.000001', -- дё»еә“ж—Ҙеҝ—ж–Үд»¶еҗҚ
  MASTER_LOG_POS=12345;              -- дё»еә“ж—Ҙеҝ—дҪҚзҪ®

    д»ҘдёҠ MASTER_LOG_FILE е’Ң MASTER_LOG_POS еҸҜйҖҡиҝҮеңЁдё»еә“иҝҗиЎҢ SHOW MASTER STATUS; иҺ·еҸ–гҖӮ

    1. еҗҜеҠЁеӨҚеҲ¶ еңЁд»Һеә“ B е’Ңд»Һеә“ C дёҠеҲҶеҲ«жү§иЎҢпјҡ 

      START SLAVE;

    2. жЈҖжҹҘеӨҚеҲ¶зҠ¶жҖҒ еңЁд»Һеә“ B е’Ңд»Һеә“ C дёҠеҲҶеҲ«жү§иЎҢпјҡ 

      SHOW SLAVE STATUS\G;

    зЎ®и®Өд»ҘдёӢдёӨйЎ№еқҮдёә Yesпјҡ - Slave_IO_Running: Yes - Slave_SQL_Running: Yes

    4. жөӢиҜ•еӨҡд»Һеә“еӨҚеҲ¶

    1. еңЁдё»еә“пјҲAпјүжү§иЎҢеҶҷж“ҚдҪңпјҡ 

      USE test_db;
INSERT INTO test_table (name, value) VALUES ('test', 123);

    2. еңЁд»Һеә“ B е’Ңд»Һеә“ C жҹҘиҜўж•°жҚ®пјҢзЎ®дҝқеҗҢжӯҘжҲҗеҠҹпјҡ 

      USE test_db;
SELECT * FROM test_table;

    5. жіЁж„ҸдәӢйЎ№

    1. д»Һеә“зҡ„ server-id е”ҜдёҖпјҡжҜҸдёӘд»Һеә“еҝ…йЎ»жңүе”ҜдёҖзҡ„ server-idгҖӮ
    2. иҙҹиҪҪеқҮиЎЎпјҡ
    3. иҜ»ж“ҚдҪңеҸҜд»ҘеҲҶеҸ‘еҲ°еӨҡдёӘд»Һеә“пјҢйҒҝе…ҚжҹҗдёҖд»Һеә“иҝҮиҪҪгҖӮ
    4. дҪҝз”Ёдёӯй—ҙд»¶пјҲеҰӮ ProxySQL жҲ– Mycatпјүз®ЎзҗҶеӨҡдёӘд»Һеә“гҖӮ
    5. ж•°жҚ®дёҖиҮҙжҖ§пјҡдё»д»ҺеӨҚеҲ¶еӯҳеңЁе»¶иҝҹпјҢйҮҚиҰҒиҜ»ж“ҚдҪңеҸҜејәеҲ¶д»Һдё»еә“иҜ»еҸ–гҖӮ
    6. д»Һеә“еҸӘиҜ»жЁЎејҸпјҡејҖеҗҜ read_only еҸӮж•°пјҢйҳІжӯўиҜҜеҶҷд»Һеә“гҖӮ
    7. еҚҠеҗҢжӯҘеӨҚеҲ¶пјҲеҸҜйҖүпјүпјҡжҸҗй«ҳж•°жҚ®дёҖиҮҙжҖ§пјҢеҸҜеңЁдё»д»Һд№Ӣй—ҙеҗҜз”ЁеҚҠеҗҢжӯҘеӨҚеҲ¶гҖӮ

    еҚҠеҗҢжӯҘеӨҚеҲ¶

    дё»д»Һд№Ӣй—ҙеҗҜз”Ё еҚҠеҗҢжӯҘеӨҚеҲ¶ жҳҜдёҖз§Қеўһејә MySQL ж•°жҚ®дёҖиҮҙжҖ§зҡ„ж–№жі•пјҢз”ЁдәҺеҮҸе°‘дё»д»ҺеӨҚеҲ¶дёӯзҡ„ж•°жҚ®дёўеӨұйЈҺйҷ©гҖӮ

    д»Җд№ҲжҳҜеҚҠеҗҢжӯҘеӨҚеҲ¶пјҹ

    еңЁй»ҳи®Өзҡ„ MySQL дё»д»ҺеӨҚеҲ¶дёӯпјҢйҮҮз”Ёзҡ„жҳҜејӮжӯҘеӨҚеҲ¶жЁЎејҸпјҡ - дё»еә“жү§иЎҢдәӢеҠЎжҸҗдәӨеҗҺпјҢдјҡз«ӢеҚіеҗ‘е®ўжҲ·з«Ҝиҝ”еӣһвҖңжҸҗдәӨжҲҗеҠҹвҖқгҖӮ - дё»еә“йҡҸеҗҺе°ҶдәҢиҝӣеҲ¶ж—Ҙеҝ— (binlog) еҸ‘йҖҒз»ҷд»Һеә“пјҢдҪҶдё»еә“е№¶дёҚзӯүеҫ…д»Һеә“зЎ®и®ӨжҳҜеҗҰж”¶еҲ°жҲ–еә”з”Ёиҝҷдәӣж—Ҙеҝ—гҖӮ

    иҝҷз§Қж–№ејҸиҷҪз„¶ж•ҲзҺҮй«ҳпјҢдҪҶжңүжҪңеңЁйЈҺйҷ©пјҡ 1. ж•°жҚ®дёўеӨұпјҡеҰӮжһңдё»еә“еҙ©жәғпјҢйғЁеҲҶе·Із»ҸжҸҗдәӨзҡ„дәӢеҠЎеҸҜиғҪе°ҡжңӘдј иҫ“еҲ°д»Һеә“пјҢеҜјиҮҙж•°жҚ®дёўеӨұгҖӮ 2. ж•°жҚ®дёҚдёҖиҮҙпјҡдё»д»Һеә“д№Ӣй—ҙеҸҜиғҪзҹӯжҡӮеӨ„дәҺдёҚеҗҢжӯҘзҠ¶жҖҒгҖӮ

    еҚҠеҗҢжӯҘеӨҚеҲ¶ жҳҜдёәдәҶи§ЈеҶіиҝҷдәӣй—®йўҳзҡ„дёҖз§Қж”№иҝӣпјҡ - дё»еә“еңЁжҸҗдәӨдәӢеҠЎж—¶пјҢдёҚд»…еҶҷе…Ҙжң¬ең°зҡ„дәҢиҝӣеҲ¶ж—Ҙеҝ—пјҢиҝҳйңҖзӯүеҫ…иҮіе°‘дёҖдёӘд»Һеә“зЎ®и®ӨжҺҘж”¶еҲ°иҜҘж—Ҙеҝ—гҖӮ - еҸӘжңүеҪ“иҮіе°‘дёҖдёӘд»Һеә“зЎ®и®ӨжҺҘж”¶еҗҺпјҢдё»еә“жүҚдјҡеҗ‘е®ўжҲ·з«Ҝиҝ”еӣһвҖңжҸҗдәӨжҲҗеҠҹвҖқгҖӮ - еҰӮжһңд»Һеә“жңӘеҸҠж—¶е“Қеә”пјҢдё»еә“дјҡеҲҮеӣһејӮжӯҘеӨҚеҲ¶жЁЎејҸд»ҘдҝқиҜҒжҖ§иғҪгҖӮ

    еҚҠеҗҢжӯҘеӨҚеҲ¶зҡ„дјҳзјәзӮ№

    дјҳзӮ№пјҡ

    1. ж•°жҚ®дёҖиҮҙжҖ§пјҡдё»д»ҺеӨҚеҲ¶зҡ„дәӢеҠЎж—Ҙеҝ—еңЁжҸҗдәӨж—¶е·Ідј иҫ“еҲ°д»Һеә“пјҢеҮҸе°‘дәҶдё»еә“еҙ©жәғж—¶зҡ„ж•°жҚ®дёўеӨұгҖӮ
    2. й«ҳеҸҜйқ жҖ§пјҡйҖӮеҗҲеҜ№ж•°жҚ®дёҖиҮҙжҖ§иҰҒжұӮиҫғй«ҳзҡ„дёҡеҠЎеңәжҷҜпјҢеҰӮйҮ‘иһҚгҖҒз”өе•Ҷзӯүе…ій”®зі»з»ҹгҖӮ

    зјәзӮ№пјҡ

    1. жҖ§иғҪеҪұе“Қпјҡдё»еә“йңҖиҰҒзӯүеҫ…д»Һеә“зЎ®и®ӨпјҢдјҡеўһеҠ дәӢеҠЎе»¶иҝҹгҖӮ
    2. дҫқиө–зҪ‘з»ңиҙЁйҮҸпјҡеҰӮжһңдё»д»Һд№Ӣй—ҙзҪ‘з»ңдёҚзЁіе®ҡпјҢеҸҜиғҪеҜјиҮҙдәӢеҠЎеӨ„зҗҶйҖҹеәҰжҳҫи‘—дёӢйҷҚгҖӮ

    еҚҠеҗҢжӯҘеӨҚеҲ¶зҡ„е·ҘдҪңжөҒзЁӢ

    1. е®ўжҲ·з«ҜжҸҗдәӨдәӢеҠЎпјҡ
    2. дё»еә“е°ҶдәӢеҠЎж—Ҙеҝ—еҶҷе…ҘдәҢиҝӣеҲ¶ж—Ҙеҝ— (binlog)гҖӮ
    3. еҸ‘йҖҒж—Ҙеҝ—з»ҷд»Һеә“пјҡ
    4. дё»еә“е°Ҷ binlog жҺЁйҖҒеҲ°д»Һеә“гҖӮ
    5. зӯүеҫ…д»Һеә“зЎ®и®Өпјҡ
    6. иҮіе°‘дёҖдёӘд»Һеә“жҺҘж”¶еҲ° binlog е№¶еҶҷе…Ҙдёӯз»§ж—Ҙеҝ—еҗҺпјҢеҗ‘дё»еә“еҸ‘йҖҒзЎ®и®Ө (ACK)гҖӮ
    7. дё»еә“иҝ”еӣһжҸҗдәӨжҲҗеҠҹпјҡ
    8. дё»еә“ж”¶еҲ°зЎ®и®ӨеҗҺпјҢеҗ‘е®ўжҲ·з«Ҝиҝ”еӣһвҖңжҸҗдәӨжҲҗеҠҹвҖқгҖӮ
    9. еҰӮжһңи¶…ж—¶жңӘж”¶еҲ°зЎ®и®ӨпјҢдё»еә“еҲҮжҚўеӣһејӮжӯҘжЁЎејҸгҖӮ

    еҰӮдҪ•еҗҜз”ЁеҚҠеҗҢжӯҘеӨҚеҲ¶

    д»ҘдёӢжҳҜй…ҚзҪ®дё»д»Һеә“еҗҜз”ЁеҚҠеҗҢжӯҘеӨҚеҲ¶зҡ„жӯҘйӘӨпјҡ

    1. жЈҖжҹҘе№¶е®үиЈ…жҸ’д»¶

    еңЁдё»еә“е’Ңд»Һеә“дёҠзЎ®дҝқе·Іе®үиЈ… rpl_semi_sync_master е’Ң rpl_semi_sync_slave жҸ’д»¶пјҡ

    еңЁдё»еә“дёҠпјҡ 

    INSTALL PLUGIN rpl_semi_sync_master SONAME 'semisync_master.so';

    еңЁд»Һеә“дёҠпјҡ 

    INSTALL PLUGIN rpl_semi_sync_slave SONAME 'semisync_slave.so';

    2. й…ҚзҪ®дё»еә“

    еңЁдё»еә“зҡ„ MySQL й…ҚзҪ®ж–Үд»¶пјҲ/etc/my.cnfпјүдёӯж·»еҠ д»ҘдёӢеҶ…е®№пјҡ 

    [mysqld]
plugin-load-add=rpl_semi_sync_master=semisync_master.so
rpl_semi_sync_master_enabled=1
rpl_semi_sync_master_timeout=1000   # и®ҫзҪ®и¶…ж—¶ж—¶й—ҙ (жҜ«з§’)

    йҮҚеҗҜ MySQL жңҚеҠЎпјҡ 

    sudo systemctl restart mysqld

    еңЁдё»еә“еҗҜз”ЁеҚҠеҗҢжӯҘеӨҚеҲ¶пјҡ 

    SET GLOBAL rpl_semi_sync_master_enabled = 1;

    3. й…ҚзҪ®д»Һеә“

    еңЁд»Һеә“зҡ„ MySQL й…ҚзҪ®ж–Үд»¶дёӯж·»еҠ д»ҘдёӢеҶ…е®№пјҡ 

    [mysqld]
plugin-load-add=rpl_semi_sync_slave=semisync_slave.so
rpl_semi_sync_slave_enabled=1

    йҮҚеҗҜд»Һеә“ MySQL жңҚеҠЎпјҡ 

    sudo systemctl restart mysqld

    еңЁд»Һеә“еҗҜз”ЁеҚҠеҗҢжӯҘеӨҚеҲ¶пјҡ 

    SET GLOBAL rpl_semi_sync_slave_enabled = 1;

    4. йӘҢиҜҒй…ҚзҪ®

    еңЁдё»еә“жү§иЎҢд»ҘдёӢе‘Ҫд»ӨпјҢжҹҘзңӢжҳҜеҗҰеҗҜз”ЁдәҶеҚҠеҗҢжӯҘеӨҚеҲ¶пјҡ 

    SHOW VARIABLES LIKE 'rpl_semi%';

    жЈҖжҹҘжҳҜеҗҰжңүд»ҘдёӢеҸҳйҮҸеҗҜз”Ёпјҡ - rpl_semi_sync_master_enabledпјҡжҳҜеҗҰеҗҜз”ЁдәҶдё»еә“зҡ„еҚҠеҗҢжӯҘгҖӮ - rpl_semi_sync_master_statusпјҡеҪ“еүҚжҳҜеҗҰдҪҝз”ЁеҚҠеҗҢжӯҘеӨҚеҲ¶гҖӮ

    йҖӮз”ЁеңәжҷҜ

    1. еҜ№ж•°жҚ®дёҖиҮҙжҖ§иҰҒжұӮй«ҳзҡ„зі»з»ҹпјҡ
    2. йҮ‘иһҚдәӨжҳ“гҖҒеә“еӯҳз®ЎзҗҶгҖҒеңЁзәҝж”Ҝд»ҳзӯүеңәжҷҜгҖӮ
    3. еҜ№жҖ§иғҪиҰҒжұӮзӣёеҜ№ж¬ЎиҰҒзҡ„зҺҜеўғпјҡ
    4. еҰӮжһңзі»з»ҹжҖ§иғҪиҰҒжұӮй«ҳпјҢеҸҜиғҪйңҖиҰҒжқғиЎЎдәӢеҠЎе»¶иҝҹй—®йўҳгҖӮ

    жіЁж„ҸдәӢйЎ№

    1. д»Һеә“ж•°йҮҸе»әи®®пјҡеҚідҪҝеҗҜз”ЁеҚҠеҗҢжӯҘеӨҚеҲ¶пјҢе»әи®®й…ҚзҪ®еӨҡдёӘд»Һеә“пјҢжҸҗй«ҳеҸҜйқ жҖ§гҖӮ
    2. зҪ‘з»ңиҙЁйҮҸиҰҒжұӮпјҡдё»д»Һд№Ӣй—ҙзҪ‘з»ңйңҖиҰҒзЁіе®ҡпјҢеҗҰеҲҷеҸҜиғҪйў‘з№ҒеӣһйҖҖеҲ°ејӮжӯҘжЁЎејҸгҖӮ
    3. йҖӮеҪ“и°ғж•ҙи¶…ж—¶ж—¶й—ҙпјҡйҖҡиҝҮеҸӮж•° rpl_semi_sync_master_timeout и®ҫзҪ®дё»еә“зӯүеҫ…д»Һеә“зЎ®и®Өзҡ„и¶…ж—¶ж—¶й—ҙгҖӮ
